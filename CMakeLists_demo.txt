cmake_minimum_required(VERSION 3.16)
project(VacuumControllerWaylandDemo)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find Qt5
find_package(Qt5 REQUIRED COMPONENTS Core Widgets)

# Enable Qt MOC
set(CMAKE_AUTOMOC ON)

# Create the demo executable
add_executable(VacuumControllerDemo wayland_test.cpp)

# Link Qt libraries
target_link_libraries(VacuumControllerDemo
    Qt5::Core
    Qt5::Widgets
)

# Set target properties
set_target_properties(VacuumControllerDemo PROPERTIES
    AUTOMOC ON
    CXX_STANDARD 17
    CXX_STANDARD_REQUIRED ON
)
